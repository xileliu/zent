```jsx
import { FormInputField, Form, FormStrategy, Pop, Icon } from 'zent';

function App() {
	const form = Form.useForm(FormStrategy.View);
	return (
		<Form type="horizontal" form={form}>
			<FormInputField
				name="field1"
				label="field1"
				required
				withoutError
			/>
			<FormInputField
				name="field2"
				type="field2"
				label="field2"
				required
			/>
			<Form.CombineErrors names={['field1', 'field2']} />
		</Form>
	);
}

ReactDOM.render(<App />, mountNode);
```
